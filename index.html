Project 01

Basic LED Control

Connections

LED Anode (+)GPIO 2
Resistor (220)GND

Instruction

This program turns the internal LED on and off every 1 second. It's the "Hello World" of electronics.

blink.ino 
#define LED_PIN 2   // GPIO 2

void setup() {
  pinMode(LED_PIN, OUTPUT);   // Set LED pin as output
}

void loop() {
  digitalWrite(LED_PIN, HIGH); // LED ON
  delay(1000);                // 1 second delay
  digitalWrite(LED_PIN, LOW);  // LED OFF
  delay(1000);                // 1 second delay
}

Project 03

Button Toggle Control

Connections

LED PINGPIO 2
BUTTON PINGPIO 18
CommonGND

Toggle Logic

Press once to Turn ON, press again to Turn OFF. Uses internal pull-up and debounce delay for stability.

button_toggle.ino 
#define LED_PIN     2
#define BUTTON_PIN  18

bool ledState = false;        // LED OFF initially
bool lastButtonState = HIGH; // Button not pressed

void setup() {
  pinMode(LED_PIN, OUTPUT);
  pinMode(BUTTON_PIN, INPUT_PULLUP);
}

void loop() {
  bool currentButtonState = digitalRead(BUTTON_PIN);

  // Detect button press (HIGH → LOW)
  if (lastButtonState == HIGH && currentButtonState == LOW) {
    ledState = !ledState;            // Toggle LED state
    digitalWrite(LED_PIN, ledState);
    delay(200); // Simple debounce
  }

  lastButtonState = currentButtonState;
}
Project 04

3-Button LED Toggle

Connections

LED 1, 2, 3GPIO 2, 4, 5
Button 1, 2, 3GPIO 18, 19, 21
CommonGND

Setup

Control 3 independent LEDs with 3 buttons. Each button toggles its corresponding LED. Perfect for multiple output control!

multi_toggle.ino 
#define LED1 2
#define LED2 4
#define LED3 5

#define BTN1 18
#define BTN2 19
#define BTN3 21

bool led1State = false;
bool led2State = false;
bool led3State = false;

bool lastBtn1 = HIGH;
bool lastBtn2 = HIGH;
bool lastBtn3 = HIGH;

void setup() {
  pinMode(LED1, OUTPUT);
  pinMode(LED2, OUTPUT);
  pinMode(LED3, OUTPUT);

  pinMode(BTN1, INPUT_PULLUP);
  pinMode(BTN2, INPUT_PULLUP);
  pinMode(BTN3, INPUT_PULLUP);
}

void loop() {
  bool btn1 = digitalRead(BTN1);
  bool btn2 = digitalRead(BTN2);
  bool btn3 = digitalRead(BTN3);

  // Button 1 toggle
  if (lastBtn1 == HIGH && btn1 == LOW) {
    led1State = !led1State;
    digitalWrite(LED1, led1State);
    delay(200);
  }

  // Button 2 toggle
  if (lastBtn2 == HIGH && btn2 == LOW) {
    led2State = !led2State;
    digitalWrite(LED2, led2State);
    delay(200);
  }

  // Button 3 toggle
  if (lastBtn3 == HIGH && btn3 == LOW) {
    led3State = !led3State;
    digitalWrite(LED3, led3State);
    delay(200);
  }

  lastBtn1 = btn1;
  lastBtn2 = btn2;
  lastBtn3 = btn3;
}
Project 05

Bluetooth LED Control

App Guide

1. Open "Serial Bluetooth Terminal" app.

2. Pair with: "ESP32_LED_Control".

3. Send '1' for ON | '0' for OFF.

bluetooth_led.ino 
#include "BluetoothSerial.h"

BluetoothSerial SerialBT;

#define LED_PIN 2

void setup() {
  pinMode(LED_PIN, OUTPUT);
  Serial.begin(9600);

  // Start Bluetooth with device name
  SerialBT.begin("ESP32_LED_Control");
}

void loop() {
  if (SerialBT.available()) {
    char data = SerialBT.read();

    if (data == '1') {
      digitalWrite(LED_PIN, HIGH);   // LED ON
      SerialBT.println("LED ON");
    }
    else if (data == '0') {
      digitalWrite(LED_PIN, LOW);    // LED OFF
      SerialBT.println("LED OFF");
    }
  }
}
Project 06

3-LED Bluetooth Control

Connections

LED 1GPIO 2
LED 2GPIO 4
LED 3GPIO 5
CommonGND

App Guide

Connect to: "ESP32_3LED_BT"

Commands: 1,3,5 (ON) | 2,4,6 (OFF)

bt_3led.ino 
#include "BluetoothSerial.h"

BluetoothSerial SerialBT;

#define LED1 2
#define LED2 4
#define LED3 5

void setup() {
  Serial.begin(9600);
  pinMode(LED1, OUTPUT);
  pinMode(LED2, OUTPUT);
  pinMode(LED3, OUTPUT);
  SerialBT.begin("ESP32_3LED_BT");
}

void loop() {
  if (SerialBT.available()) {
    char data = SerialBT.read();
    switch (data) {
      case '1': digitalWrite(LED1, HIGH); SerialBT.println("LED 1 ON"); break;
      case '2': digitalWrite(LED1, LOW);  SerialBT.println("LED 1 OFF"); break;
      case '3': digitalWrite(LED2, HIGH); SerialBT.println("LED 2 ON"); break;
      case '4': digitalWrite(LED2, LOW);  SerialBT.println("LED 2 OFF"); break;
      case '5': digitalWrite(LED3, HIGH); SerialBT.println("LED 3 ON"); break;
      case '6': digitalWrite(LED3, LOW);  SerialBT.println("LED 3 OFF"); break;
    }
  }
}

web_led_server.ino 
#include <WiFi.h>
#include <WebServer.h>

// WiFi Hotspot credentials
const char* ssid = "ESP32_LED_Server";
const char* password = "12345678";

#define LED_PIN 2

WebServer server(80);

// HTML Web Page
String webpage() {
  String html = "<!DOCTYPE html><html>";
  html += "<head><title>ESP32 LED Control</title>";
  html += "<style>";
  html += "body{font-family:Arial;text-align:center;margin-top:50px;}";
  html += "button{padding:20px 40px;font-size:20px;margin:10px;}";
  html += "</style></head>";
  html += "<body>";
  html += "<h1>ESP32 LED Control</h1>";
  html += "<a href='/on'><button>LED ON</button></a>";
  html += "<a href='/off'><button>LED OFF</button></a>";
  html += "</body></html>";
  return html;
}

// Root page
void handleRoot() {
  server.send(200, "text/html", webpage());
}

// LED ON
void handleOn() {
  digitalWrite(LED_PIN, HIGH);
  server.send(200, "text/html", webpage());
}

// LED OFF
void handleOff() {
  digitalWrite(LED_PIN, LOW);
  server.send(200, "text/html", webpage());
}

void setup() {
  Serial.begin(9600);
  pinMode(LED_PIN, OUTPUT);

  // Start ESP32 as Access Point
  WiFi.softAP(ssid, password);
  Serial.println("ESP32 Hotspot Started");
  Serial.print("IP Address: ");
  Serial.println(WiFi.softAPIP());

  // Web server routes
  server.on("/", handleRoot);
  server.on("/on", handleOn);
  server.on("/off", handleOff);

  server.begin();
}

void loop() {
  server.handleClient();
}

Project 11

IR Sensor Toggle LED

Setup

VCC -> 3.3V | GND -> GND

IR_SENSOR -> GPIO 15 | LED_PIN -> GPIO 2

ir_toggle_led.ino 
#define IR_SENSOR 15
#define LED_PIN 2

bool ledState = false;
bool lastSensorState = HIGH;

void setup() {
  pinMode(IR_SENSOR, INPUT);
  pinMode(LED_PIN, OUTPUT);
  Serial.begin(9600);
}

void loop() {
  bool sensorState = digitalRead(IR_SENSOR);

  // Detect only NEW object (edge detection)
  if (sensorState == LOW && lastSensorState == HIGH) {
    ledState = !ledState;              // TOGGLE LED
    digitalWrite(LED_PIN, ledState);
    delay(300); // simple debounce
  }

  lastSensorState = sensorState;
}

Project 14

Proximity Alarm

Rule

If object distance < 5cm, LED 2 blinks fast.

proximity_alarm.ino 
void setup() {
  pinMode(5, OUTPUT); pinMode(18, INPUT);
  pinMode(2, OUTPUT);
}

void loop() {
  digitalWrite(5, LOW); delayMicroseconds(2);
  digitalWrite(5, HIGH); delayMicroseconds(10);
  digitalWrite(5, LOW);
  long dur = pulseIn(18, HIGH);
  int dist = dur * 0.034 / 2;
  
  if (dist < 5) { // Under 5cm
    digitalWrite(2, HIGH); delay(100);
    digitalWrite(2, LOW);  delay(100);
  } else {
    digitalWrite(2, LOW);
  }
}
Project 15

DHT11 Temperature

Setup

VCC -> 3.3V | GND -> GND

DATA -> GPIO 4

Info

Reads Temp & Humidity every 2 seconds.

Requires DHT Library by Adafruit.

dht11_test.ino 
#include "DHT.h"

#define DHTPIN 4      // GPIO where DHT11 is connected
#define DHTTYPE DHT11

DHT dht(DHTPIN, DHTTYPE);

void setup() {
  Serial.begin(9600);
  dht.begin();
}

void loop() {
  float humidity = dht.readHumidity();
  float temperature = dht.readTemperature(); // Celsius

  if (isnan(humidity) || isnan(temperature)) {
    Serial.println("Failed to read from DHT sensor!");
    return;
  }

  Serial.print("Temperature: ");
  Serial.print(temperature);
  Serial.println(" °C");

  Serial.print("Humidity: ");
  Serial.print(humidity);
  Serial.println(" %");

  Serial.println("---------------------");
  delay(2000);
}
